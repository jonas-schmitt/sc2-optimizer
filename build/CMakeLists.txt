CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )
PROJECT( sc2-optimizer )


FILE( GLOB_RECURSE PROJ_SOURCES ../src/*.cpp )
FILE( GLOB_RECURSE PROJ_HEADERS ../include/*.h )

FIND_PACKAGE( MPI REQUIRED )

INCLUDE_DIRECTORIES( ${MPI_INCLUDE_PATH} )

ADD_EXECUTABLE(opt ${PROJ_SOURCES} ${PROJ_HEADERS})

TARGET_LINK_LIBRARIES( opt ${MPI_LIBRARIES} )


SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -g -fopenmp -std=c++11")
